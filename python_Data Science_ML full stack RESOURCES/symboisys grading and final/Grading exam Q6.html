legendary_count = df.groupby(['Type 1', 'Legendary']).size().unstack()
legendary_count.plot(kind='bar', stacked=True, figsize=(12,6))

plt.title('Count of Legendary and Non-Legendary Pokemon by Type', fontsize=18)
plt.xlabel('Type 1', fontsize=14)
plt.ylabel('Count', fontsize=14)
plt.xticks(rotation=90)

plt.show()
